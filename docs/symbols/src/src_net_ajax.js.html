1 /*
  2  * <ul>
  3  *  <li>uri: target uri</li>
  4  *  <li>method: get, post, ...</li>
  5  *  <li>data: data as JSON to be passed with the request</li>
  6  *  <li>success_callback(data): will be called when request was successful</li>
  7  *  <li>failure_callback(status_code, status_text, data): will be called when request was not successful</li>
  8  *  <li>complete_callback(): will be called when the request has been made</li>
  9  * </ul>
 10  * 
 11  */
 12 BetaJS.Class.extend("BetaJS.Net.AbstractAjax", {
 13 	
 14 	constructor: function (options) {
 15 		this._inherited(BetaJS.Net.AbstractAjax, "constructor");
 16 		this.__options = BetaJS.Objs.extend({
 17 			"method": "GET",
 18 			"data": {}
 19 		}, options);
 20 	},
 21 	
 22 	syncCall: function (options) {
 23 		var opts = BetaJS.Objs.clone(this.__options, 1);
 24 		opts = BetaJS.Objs.extend(opts, options);
 25 		var success_callback = opts.success_callback;
 26 		delete opts["success_callback"];
 27 		var failure_callback = opts.failure_callback;
 28 		delete opts["failure_callback"];
 29 		var complete_callback = opts.complete_callback;
 30 		delete opts["complete_callback"];
 31 		try {
 32 			var result = this._syncCall(opts);
 33 			if (success_callback)
 34 				success_callback(result);
 35 			if (complete_callback)
 36 				complete_callback();
 37 			return result;
 38 		} catch (e) {
 39 			e = BetaJS.Exceptions.ensure(e);
 40 			e.assert(BetaJS.Net.AjaxException);
 41 			if (failure_callback)
 42 				failure_callback(e.status_code(), e.status_text(), e.data());
 43 			else
 44 				throw e;
 45 		}
 46 		return false;
 47 	},
 48 	
 49 	asyncCall: function (options) {
 50 		var opts = BetaJS.Objs.clone(this.__options, 1);
 51 		opts = BetaJS.Objs.extend(opts, options);
 52 		var success_callback = opts.success_callback;
 53 		delete opts["success_callback"];
 54 		var failure_callback = opts.failure_callback;
 55 		delete opts["failure_callback"];
 56 		var complete_callback = opts.complete_callback;
 57 		delete opts["complete_callback"];
 58 		try {
 59 			var result = this._asyncCall(BetaJS.Objs.extend({
 60 				"success": function (data) {
 61 					if (success_callback)
 62 						success_callback(data);
 63 					if (complete_callback)
 64 						complete_callback();
 65 				},
 66 				"failure": function (status_code, status_text, data) {
 67 					if (failure_callback)
 68 						failure_callback(status_code, status_text, data);
 69 					else
 70 						throw new BetaJS.Net.AjaxException(status_code, status_text, data);
 71 					if (complete_callback)
 72 						complete_callback();
 73 				}
 74 			}, opts));
 75 			return result;
 76 		} catch (e) {
 77 			e = BetaJS.Exceptions.ensure(e);
 78 			e.assert(BetaJS.Net.AjaxException);
 79 			if (failure_callback)
 80 				failure_callback(e.status_code(), e.status_text(), e.data());
 81 			else
 82 				throw e;
 83 		}
 84 		return false;
 85 	},
 86 	
 87 	call: function (options) {
 88 		if (!("async" in options))
 89 			return false;
 90 		var async = options["async"];
 91 		delete options["async"];
 92 		return async ? this.asyncCall(options) : this.syncCall(options);
 93 	},
 94 	
 95 	_syncCall: function (options) {},
 96 	
 97 	_asyncCall: function (options) {}
 98 	
 99 });
