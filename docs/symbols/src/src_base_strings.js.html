1 /** @class */
  2 BetaJS.Strings = {
  3 	
  4     /** Converts a string new lines to html <br /> tags
  5      * 
  6      * @param s string
  7      * @return string with new lines replaced by <br /> 
  8      */
  9 	nl2br: function (s) {
 10 		return (s + "").replace(/([^>\r\n]?)(\r\n|\n\r|\r|\n)/g, '$1<br />$2');
 11 	},
 12 	
 13     /** Converts special characters in a string to html entitiy symbols
 14      * 
 15      * @param s string
 16      * @return converted string
 17      */
 18 	htmlentities: function (s) {
 19 		return (s + "").
 20 			replace(/&/g, '&').
 21 			replace(/</g, '<').
 22 			replace(/>/g, '>').
 23 			replace(/"/g, '"').
 24 			replace(/'/g, ''').
 25 			replace(/\//g, '/');
 26 	},
 27 	
 28 	JS_ESCAPES: {
 29 		"'":      "'",
 30 		'\\':     '\\',
 31 		'\r':     'r',
 32 		'\n':     'n',
 33 		'\t':     't',
 34 		'\u2028': 'u2028',
 35 		'\u2029': 'u2029'
 36 	},
 37 	
 38 	JS_ESCAPER_REGEX: function () {
 39 		if (!this.JS_ESCAPER_REGEX_CACHED)
 40 			this.JS_ESCAPER_REGEX_CACHED = new RegExp(BetaJS.Objs.keys(this.JS_ESCAPES).join("|"), 'g');
 41 		return this.JS_ESCAPER_REGEX_CACHED;
 42 	},
 43 	
 44     /** Converts string such that it can be used in javascript by escaping special symbols
 45      * 
 46      * @param s string
 47      * @return escaped string
 48      */
 49 	js_escape: function (s) {
 50 		var self = this;
 51 		return s.replace(this.JS_ESCAPER_REGEX(), function(match) {
 52 			return '\\' + self.JS_ESCAPES[match];
 53 		});
 54 	},
 55 	
 56     /** Determines whether a string starts with a sub string
 57      * 
 58      * @param s string in question
 59      * @param needle sub string
 60      * @return true if string in question starts with sub string
 61      */
 62 	starts_with: function (s, needle) {
 63 		return s.substring(0, needle.length) == needle;
 64 	},
 65 	
 66     /** Determines whether a string ends with a sub string
 67      * 
 68      * @param s string in question
 69      * @param needle sub string
 70      * @return true if string in question ends with sub string
 71      */
 72 	ends_with: function(s, needle) {
 73     	return s.indexOf(needle, s.length - needle.length) !== -1;
 74 	},
 75 	
 76     /** Removes sub string from a string if string starts with sub string
 77      * 
 78      * @param s string in question
 79      * @param needle sub string
 80      * @return string without sub string if it starts with sub string otherwise it returns the original string
 81      */
 82 	strip_start: function (s, needle) {
 83 		return this.starts_with(s, needle) ? s.substring(needle.length) : s;
 84 	},
 85 	
 86     /** Returns the complete remaining part of a string after a the last occurrence of a sub string
 87      * 
 88      * @param s string in question
 89      * @param needle sub string
 90      * @return remaining part of the string in question after the last occurrence of the sub string
 91      */
 92 	last_after: function (s, needle) {
 93 		return s.substring(s.lastIndexOf(needle) + needle.length, s.length);
 94 	},
 95 	
 96 	EMAIL_ADDRESS_REGEX: /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/,
 97 	
 98     /** Determines whether a string is a syntactically valid email address
 99      * 
100      * @param s string in question
101      * @return true if string looks like an email address
102      */
103 	is_email_address: function (s) {
104 		return this.EMAIL_ADDRESS_REGEX.test(s);
105 	}
106 
107 };
