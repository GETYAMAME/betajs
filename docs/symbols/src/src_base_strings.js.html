107 	 * @param html string containing html

110 	strip_html : function(html) {

112 		for ( i = 0; i < this.STRIP_HTML_TAGS.length; ++i)

118 	/** Trims all trailing and leading whitespace and removes block indentations
119 	 *
120 	 * @param s string
121 	 * @return string with trimmed whitespaces and removed block indentation
122 	 */
123 	nltrim : function(s) {
124 		var a = s.replace(/\t/g, "  ").split("\n");
125 		var len = null;
126 		var i = 0;
127 		for ( i = 0; i < a.length; ++i) {
128 			var j = 0;
129 			while (j < a[i].length) {
130 				if (a[i].charAt(j) != ' ')
131 					break;
132 				++j;
133 			}
134 			if (j < a[i].length)
135 				len = len === null ? j : Math.min(j, len);
136 		}
137 		for ( i = 0; i < a.length; ++i)
138 			a[i] = a[i].substring(len);
139 		return a.join("\n").trim();
140 	},
