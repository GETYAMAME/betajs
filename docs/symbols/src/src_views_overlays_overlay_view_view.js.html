1 BetaJS.Views.View.extend("BetaJS.Views.OverlayView", {
  2 	
  3 	_templates: {
  4 		"default": BetaJS.Templates.Cached["overlay-view-template"]

  7 	/*
  8 	 * overlay_inner (optional) : sub view to be bound to the overlay
  9 	 * 
 10 	 * anchor: "none" | "absolute" | "element" | "relative"
 11      *
 12 	 * overlay_left
 13 	 * overlay_top
 14      *
 15 	 * overlay_align_vertical: "top" | "center" | "bottom"
 16 	 * overlay_align_horizontal: "left" | "center" | "right"
 17 	 *
 18 	 * element-align-vertical: "top" | "center" | "bottom"
 19 	 * element-align-horizontal: "left" | "center" | "right"
 20      *
 21 	 * element
 22 	 */
 23 	constructor: function (options) {
 24 		options = options || {};
 25 		options.anchor = options.anchor || "absolute";
 26 		//options.hide_on_leave = "hide_on_leave" in options ? options.hide_on_leave : true;
 27 		options.visible = "visible" in options ? options.visible : false;
 28 		options.children_classes = options.children_classes || [];
 29 		if (BetaJS.Types.is_string(options.children_classes))
 30 			options.children_classes = options.children_classes.split(" ");
 31 		options.children_classes.push("overlay-container-class");
 32 		this._inherited(BetaJS.Views.OverlayView, "constructor", options);
 33 		this._setOption(options, "anchor", "none");
 34 		this._setOption(options, "element", "");
 35 		this._setOption(options, "overlay_left", 0);
 36 		this._setOption(options, "overlay_top", 0);
 37 		this._setOption(options, "overlay_align_vertical", "top");
 38 		this._setOption(options, "overlay_align_horizontal", "left");
 39 		this._setOption(options, "element_align_vertical", "bottom");
 40 		this._setOption(options, "element_align_horizontal", "left");
 41 		if (options.overlay_inner) {
 42 			options.overlay_inner.setEl('[data-selector="container"]');
 43 			this.overlay_inner = this.addChild(options.overlay_inner);
 44 		}
 45 		if (!("hide_on_leave" in options) || options.hide_on_leave)
 46 			this.add_module(BetaJS.Views.Modules.HideOnLeave.singleton());
 47 		this.on("show", this._after_show, this);
 48 	},
 49 	
 50 	_after_show: function () {	
 51 		if (this.__anchor == "none")
 52 			return;
 53 		var overlay = this.$(".overlay-container-class");
 54 		var width = overlay.outerWidth();
 55 		var height = overlay.outerHeight();
 56 
 57 		var left = this.__overlay_left;
 58 		var top = this.__overlay_top;
 59 		
 60 		if (this.__overlay_align_vertical == "bottom")
 61 			top -= height;
 62 		else if (this.__overlay_align_vertical == "center")
 63 			top -= Math.round(height/2);
 64 
 65 		if (this.__overlay_align_horizontal == "right")
 66 			left -= width;
 67 		else if (this.__overlay_align_horizontal == "center")
 68 			left -= Math.round(width/2);
 69 			
 70 		var element = this.$el;
 71 		if (this.__anchor == "element" && this.__element) {
 72 			element = this.__element;
 73 			if (BetaJS.Types.is_string(element))
 74 				element = BetaJS.$(element);
 75 			else if (BetaJS.Class.is_class_instance(element))
 76 				element = element.$el;
 77 		}
 78 		if (this.__anchor == "relative" || this.__anchor == "element") {
 79 			element_width = element.outerWidth();
 80 			element_height = element.outerHeight();
 81 			left += element.offset().left - $(window).scrollLeft();
 82 			top += element.offset().top - $(window).scrollTop();
 83 			if (this.__element_align_vertical == "bottom")
 84 				top += element_height;
 85 			else if (this.__element_align_vertical == "center")
 86 				top += Math.round(element_height/2);
 87 			if (this.__element_align_horizontal == "right")
 88 				left += element_width;
 89 			else if (this.__element_align_horizontal == "center")
 90 				left += Math.round(element_width/2);
 91 		}
 92 		overlay.css("left", left + "px");
 93 		overlay.css("top", top + "px");
 94 	}
