 19 		var query = constrained_query.query;
 20 		var options = constrained_query.options;

 24 			execute_options.sort = options.sort;

 28 			execute_query = query;

 30 				if ("skip" in options && "skip" in query_capabilities)
 31 					execute_options.skip = options.skip;
 32 				if ("limit" in options && "limit" in query_capabilities)
 33 					execute_options.limit = options.limit;
 34 			}
 35 		}
 36 		var params = [execute_query, execute_options];
 37 		if (callbacks)

 39 		var success_call = function (raw) {

 41 			if (raw == null)
 42 				iter = new BetaJS.Iterators.ArrayIterator([])
 43 			else if (BetaJS.Types.is_array(raw))
 44 				iter = new BetaJS.Iterators.ArrayIterator(raw);		
 45 			if (!("query" in query_capabilities || BetaJS.Types.is_empty(query)))
 46 				iter = new BetaJS.Iterators.FilteredIterator(iter, function(row) {
 47 					return BetaJS.Queries.evaluate(query, row);
 48 				});
 49 			if ("sort" in options && !("sort" in execute_options))
 50 				iter = new BetaJS.Iterators.SortedIterator(iter, BetaJS.Comparators.byObject(options.sort));
 51 			if ("skip" in options && !("skip" in execute_options))
 52 				iter = new BetaJS.Iterators.SkipIterator(iter, options["skip"]);
 53 			if ("limit" in options && !("limit" in execute_options))
 54 				iter = new BetaJS.Iterators.LimitIterator(iter, options["limit"]);
 55 			if (callbacks && callbacks.success)
 56 				callbacks.success(iter);
 57 			return iter;
 58 		};
 59 		var exception_call = function (e) {
 60 			if (callbacks && callbacks.exception)
 61 				callbacks.exception(e)
 62 			else
 63 				throw e;
 64 		};
 65 		if (callbacks)
 66 			query_function.apply(query_context || this,[execute_query, execute_options, {success: success_call, exception: exception_call}])
 67 		else
 68 			try {
 69 				var raw = query_function.apply(query_context || this, [execute_query, execute_options]);
 70 				return success_call(raw);
 71 			} catch (e) {
 72 				exception_call(e);
 73 			}		
 74 	}
