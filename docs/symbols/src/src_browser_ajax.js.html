1 /*
  2  * <ul>
  3  *  <li>uri: target uri</li>
  4  *  <li>method: get, post, ...</li>
  5  *  <li>data: data as JSON to be passed with the request</li>
  6  *  <li>success_callback(data): will be called when request was successful</li>
  7  *  <li>failure_callback(status_code, status_text, data): will be called when request was not successful</li>
  8  *  <li>complete_callback(): will be called when the request has been made</li>
  9  * </ul>
 10  * 
 11  */
 12 BetaJS.Class.extend("BetaJS.Browser.AbstractAjax", {
 13 	
 14 	constructor: function (options) {
 15 		this._inherited(BetaJS.Browser.AbstractAjax, "constructor");
 16 		this.__options = BetaJS.Objs.extend({
 17 			"method": "GET",
 18 			"data": {}
 19 		}, options);
 20 	},
 21 	
 22 	syncCall: function (options) {
 23 		var opts = BetaJS.Objs.clone(this.__options, 1);
 24 		opts = BetaJS.Objs.extend(opts, options);
 25 		var success_callback = opts.success_callback;
 26 		delete opts["success_callback"];
 27 		var failure_callback = opts.failure_callback;
 28 		delete opts["failure_callback"];
 29 		var complete_callback = opts.complete_callback;
 30 		delete opts["complete_callback"];
 31 		try {
 32 			var result = this._syncCall(opts);
 33 			if (success_callback)
 34 				success_callback(result);
 35 			if (complete_callback)
 36 				complete_callback();
 37 			return result;
 38 		} catch (e) {
 39 			e = BetaJS.Exceptions.ensure(e);
 40 			e.assert(BetaJS.Browser.AjaxException);
 41 			if (failure_callback)
 42 				failure_callback(e.status_code(), e.status_text(), e.data())
 43 			else
 44 				throw e;
 45 		}
 46 	},
 47 	
 48 	asyncCall: function (options) {
 49 		var opts = BetaJS.Objs.clone(this.__options, 1);
 50 		opts = BetaJS.Objs.extend(opts, options);
 51 		var success_callback = opts.success_callback;
 52 		delete opts["success_callback"];
 53 		var failure_callback = opts.failure_callback;
 54 		delete opts["failure_callback"];
 55 		var complete_callback = opts.complete_callback;
 56 		delete opts["complete_callback"];
 57 		try {
 58 			var result = this._asyncCall(BetaJS.Objs.extend({
 59 				"success": function (data) {
 60 					if (success_callback)
 61 						success_callback(data);
 62 					if (complete_callback)
 63 						complete_callback();
 64 				},
 65 				"failure": function (status_code, status_text, data) {
 66 					if (failure_callback)
 67 						failure_callback(status_code, status_text, data)
 68 					else
 69 						throw new BetaJS.Browser.AjaxException(status_code, status_text, data);
 70 					if (complete_callback)
 71 						complete_callback();
 72 				}
 73 			}, opts));
 74 			return result;
 75 		} catch (e) {
 76 			e = BetaJS.Exceptions.ensure(e);
 77 			e.assert(BetaJS.Browser.AjaxException);
 78 			if (failure_callback)
 79 				failure_callback(e.status_code(), e.status_text(), e.data())
 80 			else
 81 				throw e;
 82 		}
 83 	},
 84 	
 85 	call: function (options) {
 86 		if (!("async" in options))
 87 			return false;
 88 		var async = options["async"];
 89 		delete options["async"];
 90 		return async ? this.asyncCall(options) : this.syncCall(options);
 91 	},
 92 	
 93 	_syncCall: function (options) {},
 94 	
 95 	_asyncCall: function (options) {}
 96 	
 97 });

124 BetaJS.Browser.AbstractAjax.extend("BetaJS.Browser.JQueryAjax", {
125 	
126 	_syncCall: function (options) {
127 		var result;
128 		BetaJS.$.ajax({
129 			type: options.method,
130 			async: false,
131 			url: options.uri,
132 			dataType: options.decodeType ? options.decodeType : null, 
133 			data: options.encodeType && options.encodeType == "json" ? JSON.stringify(options.data) : options.data,
134 			success: function (response) {
135 				result = response;
136 			},
137 			error: function (jqXHR, textStatus, errorThrown) {
138 				throw new BetaJS.Browser.AjaxException(jqXHR.status, errorThrown, JSON.parse(jqXHR.responseText));
139 			}
140 		});
141 		return result;
142 	},
143 	
144 	_asyncCall: function (options) {
145 		BetaJS.$.ajax({
146 			type: options.method,
147 			async: true,
148 			url: options.uri,
149 			dataType: options.decodeType ? options.decodeType : null, 
150 			data: options.encodeType && options.encodeType == "json" ? JSON.stringify(options.data) : options.data,
151 			success: function (response) {
152 				options.success(response);
153 			},
154 			error: function (jqXHR, textStatus, errorThrown) {
155 				options.failure(jqXHR.status, errorThrown, JSON.parse(jqXHR.responseText));
156 			}
157 		});
158 	}
