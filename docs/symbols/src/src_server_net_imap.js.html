104 	fetch: function (options, callbacks) {
105 		options = options || {};
106 		var bodies = [];
107 		if (!("headers" in options) || options.headers)
108 			bodies.push('HEADER.FIELDS (FROM TO SUBJECT DATE)');
109 		if (!("body" in options) || options.body)
110 			bodies.push('TEXT');
111 		var seq_start = 1;
112 		var seq_end = 100;
113 		if (options.seq_count) {
114 			if (options.seq_end) {
115 				seq_end = options.seq_end;
116 				seq_start = seq_end - options.seq_count + 1;
117 			} else {
118 				seq_start = options.seq_start || seq_start;
119 				seq_end = seq_start + options.seq_count - 1;
120 			}
121 		} else {
122 			seq_start = options.seq_start || seq_start;
123 			seq_end = options.seq_end || seq_start + 99;
124 		}
125 		if (options.reverse) {
126 			var dist = seq_end - seq_start;
127 			seq_end = this.__count - seq_start + 1;
128 			seq_start = seq_end - dist;
129 		}
130 		var f = this.__imap.seq.fetch(seq_start + ":" + seq_end, {
131 			bodies : bodies,
132 			struct : true
133 		});
134 		this.__query(f, callbacks);
135 	},

172 	__parse: function (header, body, attrs) {
173 		this.trigger("parse", header, body, attrs);
174 		var mail = {};
175 		/* Attrs */
176     	mail.uid = attrs.uid;
177     	mail.threadid = attrs['x-gm-thrid'];
178     	mail.id = attrs.uid;
179     	mail.seqid = attrs.seqno;
180     	/* Header */
181     	if (header && header.subject && header.subject.length > 0)
182 			mail.subject = header.subject[0];
183     	if (header && header.to && header.to.length > 0)
184 			mail.to = header.to[0];
185     	if (header && header.from && header.from.length > 0)
186 			mail.from = header.from[0];
187     	if (header && header.date && header.date.length > 0) {
188 			var d = new Date(header.date[0]);
189 			mail.date = d.getTime();
190 		}
191 		if (body) {
192 			/* Meta Body */
193 			var struct = attrs.struct;
194 			var parts = [];
195 			if (struct.length > 1) {
196 				var boundary = struct[0].params.boundary;
197 				var rest = body;
198 				var boundary_prefix = rest.indexOf(boundary);
199 				for (var i = 1; i < struct.length; ++i) {
200 					var obj = struct[i][0] || {};
201 					// Remove everything before boundary
202 					rest = rest.substring(rest.indexOf(boundary) + boundary.length);
203 					// Remove everything before empty line
204 					rest = rest.substring(rest.indexOf("\r\n\r\n") + "\r\n\r\n".length);
205 					// Ignore attachments for now
206 					if (obj.disposition || obj.type != 'text')
207 						continue;
208 					var j = rest.indexOf(boundary) - boundary_prefix;
209 					parts.push({meta: obj, body: j >= 0 ? rest.substring(0, j) : rest});
210 				}
211 			} else
212 				parts.push({meta: struct[0], body: body});
213 			var html_body = null;
214 			var text_body = null;
215 			for (var k = 0; k < parts.length; ++k) {
216 				var encoded = parts[k].body;
217 				var encoding = parts[k].meta.encoding.toLowerCase();
218 				try {
219 					if (encoding == "quoted-printable") {
220 						encoded = this.__quoted_printable.decode(encoded).toString();
221 					} else {
222 						encoded = new Buffer(encoded, encoding).toString();
223 					}
224 				} catch (e) {}
225 				if (parts[k].meta.subtype == "html")
226 					html_body = encoded;
227 				else
228 					text_body = encoded;
229 			}
230 			if (!text_body && html_body) {
231 				text_body = this.__html_strip.html_strip(html_body, {
232 			        include_script : false,
233 			        include_style : false,
234 			        compact_whitespace : true
235 				});
236 		    }
237 			mail.html_body = html_body;
238 			mail.text_body = text_body;
239 		}
240 		return mail;
241 	}
