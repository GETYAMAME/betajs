 33 		container.removeClass("fullscreen-overlay-fit");
 34 		var outer_width = outer.outerWidth();
 35 		var outer_height = outer.outerHeight();
 36 		var inner_width = inner.outerWidth();
 37 		var inner_height = inner.outerHeight();
 38 		var left = Math.floor((outer_width - inner_width) / 2);
 39 		var top = Math.floor((outer_height - inner_height) / 2);
 40 		if (left >= 0 && top >= 0) {
 41 			container.css("left", left + "px");
 42 			container.css("top", top + "px");
 43 			container.addClass("fullscreen-overlay-float");
 44 		} else {
 45 			container.css("left", "0px");
 46 			container.css("top", "0px");

 48 		}

 54 	},
