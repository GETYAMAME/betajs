1 /** @class */
  2 BetaJS.Types = {
  3 	
  4     /** Returns whether argument is an object
  5      * 
  6      * @param x argument
  7      * @return true if x is an object
  8      */
  9 	is_object: function (x) {
 10 		return typeof x == "object";
 11 	},
 12 	
 13     /** Returns whether argument is an array
 14      * 
 15      * @param x argument
 16      * @return true if x is an array
 17      */
 18 	is_array: function (x) {
 19 		return Object.prototype.toString.call(x) === '[object Array]';
 20 	},
 21 	
 22     /** Returns whether argument is undefined (which is different from being null)
 23      * 
 24      * @param x argument
 25      * @return true if x is undefined
 26      */
 27 	is_undefined: function (x) {
 28 		return typeof x == "undefined";		
 29 	},
 30 	
 31     /** Returns whether argument is null (which is different from being undefined)
 32      * 
 33      * @param x argument
 34      * @return true if x is null
 35      */
 36 	is_null: function (x) {
 37 		return x == null;
 38 	},
 39 	
 40     /** Returns whether argument is undefined or null
 41      * 
 42      * @param x argument
 43      * @return true if x is undefined or null
 44      */
 45 	is_none: function (x) {
 46 		return this.is_undefined(x) || this.is_null(x);
 47 	},
 48 	
 49     /** Returns whether argument is defined (could be null)
 50      * 
 51      * @param x argument
 52      * @return true if x is defined
 53      */
 54 	is_defined: function (x) {
 55 		return typeof x != "undefined";
 56 	},
 57 	
 58     /** Returns whether argument is empty (undefined, null, an empty array or an empty object)
 59      * 
 60      * @param x argument
 61      * @return true if x is empty
 62      */
 63 	is_empty: function (x) {
 64 		if (this.is_none(x)) 
 65 			return true;
 66 		if (this.is_array(x))
 67 			return x.length == 0;
 68 		if (this.is_object(x)) {
 69 			for (var key in x)
 70 				return false;
 71 			return true;
 72 		}
 73 		return false; 
 74 	},
 75 	
 76     /** Returns whether argument is a string
 77      * 
 78      * @param x argument
 79      * @return true if x is a a string
 80      */
 81 	is_string: function (x) {
 82 		return typeof x == "string";
 83 	},
 84 	
 85     /** Returns whether argument is a function
 86      * 
 87      * @param x argument
 88      * @return true if x is a function
 89      */
 90 	is_function: function (x) {
 91 		return typeof x == "function";
 92 	},
 93 	
 94     /** Returns whether argument is boolean
 95      * 
 96      * @param x argument
 97      * @return true if x is boolean
 98      */
 99 	is_boolean: function (x) {
100 		return typeof x == "boolean";
101 	},
102 	
103     /** Compares two values
104      * 
105      * If values are booleans, we compare them directly.
106      * If values are arrays, we compare them recursively by their components.
107      * Otherwise, we use localeCompare which compares strings.
108      * 
109      * @param x left value
110      * @param y right value
111      * @return 1 if x > y, -1 if x < y and 0 if x == y
112      */
113 	compare: function (x, y) {
114 		if (BetaJS.Types.is_boolean(x) && BetaJS.Types.is_boolean(y))
115 			return x == y ? 0 : (x ? 1 : -1);
116 		if (BetaJS.Types.is_array(x) && BetaJS.Types.is_array(y)) {
117 			var len_x = x.length;
118 			var len_y = y.length;
119 			var len = Math.min(len_x, len_y);
120 			for (var i=0; i < len; ++i) {
121 				var c = this.compare(x[i], y[i]);
122 				if (c != 0)
123 					return c;
124 			}
125 			return len_x == len_y ? 0 : (len_x > len_y ? 1 : -1);
126 		}
127 		return x.localeCompare(y);			
128 	},
129 	
130     /** Parses a boolean string
131      * 
132      * @param x boolean as a string
133      * @return boolean value
134      */	
135 	parseBool: function (x) {
136 		if (this.is_boolean(x))
137 			return x;
138 		if (x == "true")
139 			return true;
140 		if (x == "false")
141 			return false;
142 		return null;
143 	}
144 
145 };
