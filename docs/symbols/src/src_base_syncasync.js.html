1 /** @class */
  2 BetaJS.SyncAsync = {
  3 	
  4     /** Converts a synchronous function to an asynchronous one and calls it
  5      * 
  6      * @param callbacks callbacks object with success and failure
  7      * @param syncCall the synchronous function
  8      * @param context optional object context
  9      */	
 10 	syncToAsync: function (callbacks, syncCall, context) {
 11 		try {
 12 			callbacks.success.call(callbacks.context || this, syncCall.apply(context || this));
 13 		} catch (e) {
 14 			callbacks.failure.call(callbacks.context || this, e);
 15 		}
 16 	},
 17 	
 18     /** Either calls a synchronous or asynchronous function depending on whether useSync is given
 19      * 
 20      * @param callbacks callbacks object with success and failure (or null)
 21      * @param useSync use synchronous call?
 22      * @param syncCall the synchronous function
 23      * @param asyncCall the asynchronous function
 24      * @param context optional object context
 25      * @return the function return data
 26      */	
 27 	either: function (callbacks, useSync, syncCall, asyncCall, context) {
 28 		context = context || this;
 29 		if (callbacks) {
 30 			if (useSync)
 31 				this.syncToAsync(callbacks, syncCall, context);
 32 			else
 33 				asyncCall.call(context, callbacks);
 34 		} else
 35 			return syncCall.apply(context);
 36 		return null;
 37 	},
 38 	
 39 	SYNC: 1,
 40 	ASYNC: 2,
 41 	ASYNCSINGLE: 3,
 42 	
 43 	toCallbackType: function (callbacks, type) {
 44 		if (type == this.ASYNCSINGLE)
 45 			return function (err, result) {
 46 				if (err)
 47 					callbacks.failure.call(callbacks.context || this, err);
 48 				callbacks.success.call(callbacks.context || this, result);
 49 			};
 50 		return callbacks;
 51 	},
 52 	
 53 	then: function () {
 54 		var args = BetaJS.Functions.matchArgs(arguments, {
 55 			func_ctx: "object",
 56 			func: true,
 57 			params: "array",
 58 			type: "number",
 59 			callbacks: true,
 60 			success_ctx: "object",
 61 			success: "function"
 62 		});
 63 		var func_ctx = args.func_ctx || this;
 64 		var func = args.func;
 65 		var params = args.params || [];
 66 		var callbacks = args.callbacks;
 67 		var type = args.type || (callbacks ? this.ASYNC : this.SYNC);
 68 		var success_ctx = args.success_ctx || func_ctx;
 69 		var success = args.success;
 70 		if (type != this.SYNC) {			
 71 			params.push(this.toCallbackType(success ? {
 72 				context: callbacks.context,
 73 				success: function (ret) {
 74 					success.call(success_ctx, ret, callbacks);
 75 				},
 76 				failure: callbacks.failure
 77 			} : callbacks, type));
 78 			func.apply(func_ctx, params);
 79 		} else if (callbacks) {
 80 			try {
 81 				if (success)
 82 					success.call(success_ctx, func.apply(func_ctx, params), callbacks);
 83 				else
 84 					callbacks.success.call(callbacks.context || this, func.apply(func_ctx, params));
 85 			} catch (e) {
 86 				callbacks.failure.call(callbacks.context || this, e);
 87 			}
 88 		} else {
 89 			var ret = func.apply(func_ctx, params);
 90 			if (success)
 91 				success.call(success_ctx, ret, {
 92 					success: function (retv) {
 93 						ret = retv;
 94 					},
 95 					failure: function (err) {
 96 						throw err;
 97 					}
 98 				});
 99 			return ret;
100 		}
101 		return null;
102 	}
103 
104 };
