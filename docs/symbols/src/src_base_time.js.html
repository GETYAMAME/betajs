  3 	format_time: function(t, s) {
  4 		var seconds = this.seconds(t);
  5 		var minutes = this.minutes(t);
  6 		var hours = this.hours(t);
  7 		var replacers = {
  8 			"hh": hours < 10 ? "0" + hours : hours, 
  9 			"h": hours, 
 10 			"mm": minutes < 10 ? "0" + minutes : minutes, 
 11 			"m": minutes, 
 12 			"ss": seconds < 10 ? "0" + seconds : seconds, 
 13 			"s": seconds
 14 		};
 15 		for (var key in replacers)
 16 			s = s.replace(key, replacers[key]);
 17 		return s;
 18 	},

 20 	make: function (data) {
 21 		var t = 0;
 22 		var multipliers = {
 23 			hours: 60,
 24 			minutes: 60,
 25 			seconds: 60,
 26 			milliseconds: 1000
 27 		};
 28 		for (var key in multipliers) {
 29 			t *= multipliers[key];
 30 			if (key in data)
 31 				t += data[key];
 32 		}
 33 		return t;

 72 	format_period: function (t) {
 73 		t = Math.round(t / 1000);
 74 		if (t < 60)
 75 			return t + " " + BetaJS.Locales.get(t == 1 ? "second" : "seconds");
 76 		t = Math.round(t / 60);
 77 		if (t < 60)
 78 			return t + " " + BetaJS.Locales.get(t == 1 ? "minute" : "minutes");
 79 		t = Math.round(t / 60);
 80 		if (t < 24)
 81 			return t + " " + BetaJS.Locales.get(t == 1 ? "hour" : "hours");
 82 		t = Math.round(t / 24);
 83 		return t + " " + BetaJS.Locales.get(t == 1 ? "day" : "days");
 84 	},

 87 		options = BetaJS.Objs.extend({

 92 		var d = new Date(t);
 93 		if (options.locale) {
 94 			if (options.date) {
 95 				if (options.time)
 96 					return d.toLocaleString();

 99 			} else

101 		} else {
